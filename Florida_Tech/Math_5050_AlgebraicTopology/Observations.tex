	\item \textbf{Vocabulary from Topology}
	\begin{itemize}
		\item \textbf{Properly Discontinuous}
		
		A properly discontinuous action of a group \(G\) on a topological space \(X\) is a group action where for every point \(x\in X\), there exists a neighborhood \(U\) such that the set of group elements \(g\in G\) moving \(U\) to intersect itself is limited to the identity, i.e., \(gU\cap U\ne \emptyset \) only for \(g=e\). This ensures the quotient map \(X\rightarrow X/G\) is a covering map.
		
		\item \textbf{Connected}
				
		A topological space $X$ is said to be \textit{disconnected} if it is the union of two disjoint non-empty open sets. Otherwise, $X$ is said to be \DEFINE{connected.}
		
		\item \textbf{Simply Connected}

		A topological space is simply connected if it is path-connected and every continuous loop within it can be continuously shrunk to a point (i.e., trivial group).
		
		\item \textbf{Path Connected}
		
		 A path from a point $x$ to a point $y$ in a topological space 
$X$ is a continuous function 
$f$ from the unit interval $[0,1]$ to $X$ with $f(0)=$ and $f(1)=y$. A path-component of $X$ is an equivalence class of $X$ under the equivalence relation which makes $x$ equivalent to $y$ if and only if there is a path from $x$ to $y$. The space $X$ is said to be path-connected (or pathwise connected or \textbf{0}-connected) if there is exactly one path-component.

		\item \textbf{Locally Path Connected}
		
		A space $X$ is called locally connected at $x$ if every neighborhood of $x$ contains a connected open neighborhood of $x$, that is, if the point $x$ has a neighborhood base consisting of connected open sets.
		
		\item \textbf{Semi-locally Path Connected}
		
		A space $X$ is called semi-locally simply connected if every point $x$ in $X$ and every neighborhood V of x has an open neighborhood $U$ of x such that $x\in U\subset V$ with the property that every loop in $U$ can be contracted to a single point within $X$ (i.e. every loop in $U$ is nullhomotopic in $X$).
		
		\item \textbf{Homeomorphic}
		
		Give two topological spaces $X$ and $Y$.  They are said to be \DEFINE{homeomorphic} if there exists a continuous function $f: X \to Y$ and its inverse. (think ``isomorphsim" but in topological terms).
		
		\item \textbf{Locally Homeomorphic}

	\item \textbf{Deformation Retract}
	
	\begin{definition}[Deformation Retract]
	A subspace $A$ of $X$ is \DEFINE{deformation retract} of $X$ if there is a retraction $g:X \to A$ (i.e., a map $g: X\to A$ with $g(a)=a$ for every $a\in A$)  such that $g$ is homotopic to the identiy map $id: X \to X$.  A subspace $A$ of $X$ is a \DEFINE{strong deformation retract} if there is a retraction $g: X \to A$ such that $g$ is homotopic rel $A$ to the identity map $id: X \to X$.
	
	\end{definition}
		
	\item \textbf{Homotopy Equivalence and Deformation Retract}
	
	\DEFINE{Homotopy}: Two maps $f_0:(X,A) \to (Y,B)$ and $f_1:(X,A) \to (Y,B)$ are \DEFINE{homotopic}, i.e., $f_0 \sim f_1$, if there is a map 
	\begin{align*}
		F: (X,A)\times I \to (Y,B)
	\end{align*}where $F(x,0) = f_0$ and $F(x,1) = f_1$ for every $x\in X$.  \textit{Homotopy} is an equivalence relation.

	\begin{definition}[Homotopy Equivalence] A continuous map $f: X \to Y$ is a \DEFINE{homotopy equivalence} if there exists a continuous map $g:Y \to X$ such that $f \circ g = \mathbb{I}_Y$  and $g \circ f = \mathbb{I}_X$. This means the maps are inverses of each other up to homotopy, implying that the spaces $X$ and $Y$ share the same homotopy type.
	\end{definition}

\section*{Homology}

\begin{description}
\item \textbf{Simplicial Mapping, $\Delta^n$}

The \DEFINE{simplicial mapping}, $\Delta_n(X)$, is the set of all mappings of n-simplexes onto the topological space $X$.  These mappings form a group.

\item \textbf{Chain Complex}

Given a set of nested groups $C_n$ such that 
\begin{align*}
\dots \to C_{n+1} \overset{\partial_{n+1}}\longrightarrow C_{n} \overset{\partial_{n}}\longrightarrow C_{n-1} \overset{\partial_{n-1}}\longrightarrow \dots \longrightarrow C_{1} \overset{\partial_{1}}\longrightarrow C_{0} \overset{\partial_{0}}\longrightarrow \to 0
\end{align*}Where the image of $\IMG\partial_{n+1} \subset \ker \partial_n$ for all $n$.  We define the \DEFINE{exact chain} if $\IMG\partial_{n+1} = \ker \partial_n$ for all $n$.

\item \textbf{The $n\ITH$ simplicial homology group, $H_n^\Delta(X)$}

The chain complex where $C_n = \Delta_n$ of simplicial elements.  Where each element is defined as $H_n(X) = \ker\partial_n/\IMG\partial_{n+1}$.

\item \textbf{The $n\ITH$ singular homology group, $H_n(X)$}

The chain complex where $C_n$ is the set of continuous functions on $X$ in $n$ variables.  Where each element is defined as $H_n(X) = \ker\partial_n/\IMG\partial_{n+1}$.
\end{description}

\section*{Regarding homologies}

\begin{description}
	\item Define the following
	\begin{align*}
		Z_n(X) &= \ker(\partial_n\,:\, C_n(X) \to C_{n-1}(X)), \text{ the group of \textit{singular n-cycles}}\\
		B_n(X) &= \IM(\partial_{n+1}\,:\, C_{n+1}(X) \to C_{n}(X)), \text{ the group of \textit{singular n-boundaries}}\\
		H_n(X) &= Z_n(X)/B_n(X), \text{ the $n\ITH$ \textit{singular homology group} of $X$}
	\end{align*}

	\textbf{Explain $H_n(X)$}

    \item \textbf{The Chain Group $ C_n(X) $}
To understand $ H_n(X) $, we start with the singular $ n $-chain group, $ C_n(X) $. This is the free abelian group generated by all singular $ n $-simplices in the space $ X $. An element of this group is a formal finite sum $ \sum_i n_i \sigma_i $, where each $ \sigma_i $ is a continuous map from the standard $ n $-simplex $ \Delta^n $ to $ X $.

    \item \textbf{The Boundary Operator $ \partial_n $}
The boundary operator $ \partial_n: C_n(X) \to C_{n-1}(X) $ is a homomorphism that takes an $ n $-simplex and maps it to an alternating sum of its $ (n-1) $-dimensional faces. A fundamental property of this operator is that $ \partial_n \circ \partial_{n+1} = 0 $, which means that the boundary of a boundary is always zero.

    \item \textbf{Singular $ n $-cycles $ Z_n(X) $}
The group of singular $ n $-cycles, denoted by $ Z_n(X) $, is the kernel of the boundary operator:
\[ Z_n(X) = \ker(\partial_n) = \{ c \in C_n(X) \mid \partial_n(c) = 0 \} \]
These represent chains that have no boundary (they are "closed").

    \item \textbf{Singular $ n $-boundaries $ B_n(X) $}
The group of singular $ n $-boundaries, denoted by $ B_n(X) $, is the image of the boundary operator from the dimension above:
\[ B_n(X) = \text{im}(\partial_{n+1}) = \{ c \in C_n(X) \mid c = \partial_{n+1}(d) \text{ for some } d \in C_{n+1}(X) \} \]
Because $ \partial_n \circ \partial_{n+1} = 0 $, every boundary is necessarily a cycle ($ B_n(X) \subseteq Z_n(X) $).

    \item \textbf{The Homology Group $ H_n(X) $}
The $ n $-th singular homology group $ H_n(X) $ is defined as the quotient group of cycles modulo boundaries:
\[ H_n(X) = Z_n(X) / B_n(X) \]
Intuitively, $ H_n(X) $ measures the presence of "$ n $-dimensional holes" in the space. It counts cycles that are not boundaries of any $ (n+1) $-dimensional shape. If two cycles differ by a boundary, they represent the same element (homology class) in $ H_n(X) $.
\end{description}

	\item \textbf{What are the properties of Covering Spaces?}
	
	A covering space is a concept deeply connected to the fundamental group. Let's explore its properties.\\

Let \(p: E \to B\) be a continuous surjective map between topological spaces. We call \(p\) a \DEFINE{covering map}, and \(E\) a \DEFINE{covering space} of \(B\), if it satisfies the following property:\\

For every point \(b \in B\), there exists an open neighborhood \(U\) of \(b\) in \(B\) such that its preimage \(p^{-1}(U)\) is a disjoint union of open sets \(V_i\) in \(E\), and for each \(i\), the restriction of \(p\) to \(V_i\), denoted \(p|_{V_i}: V_i \to U\), is a homeomorphism. The neighborhood \(U\) is called an \DEFINE{evenly covered} neighborhood.

 \textbf{Key Properties of Covering Spaces:}

\begin{enumerate}

\item \textbf{\textbf{Local Homeomorphism:}} A covering map \(p: E \to B\) is a local homeomorphism. This means that for every point \(e \in E\), there is a neighborhood of \(e\) that is mapped homeomorphically onto its image in \(B\). This follows directly from the definition of an evenly covered neighborhood.

\item \textbf{\textbf{Lifting Properties:}} Covering spaces have unique lifting properties for paths and homotopies. This is where the connection to the fundamental group becomes crucial.
\begin{itemize}
    \item \textbf{\textbf{Path Lifting Property:}} Given a path \(\gamma: [0, 1] \to B\) in the base space and a starting point \(e_0 \in E\) such that \(p(e_0) = \gamma(0)\), there exists a unique path \(\tilde{\gamma}: [0, 1] \to E\) in the covering space that starts at \(e_0\) (i.e., \(\tilde{\gamma}(0) = e_0\)) and covers \(\gamma\) (i.e., \(p \circ \tilde{\gamma} = \gamma\)). This lifted path \(\tilde{\gamma}\) is called a \DEFINE{lift} of \(\gamma\).
    \item \textbf{\textbf{Homotopy Lifting Property:}} If two paths in \(B\) are homotopic, their lifts in \(E\) starting at the same point will also be homotopic (specifically, they will end at the same point).
\end{itemize}

\item \textbf{\textbf{Relationship with the Fundamental Group:}} The connection between covering spaces and the fundamental group is one of the most important results in algebraic topology.
\begin{itemize}

    \item Let \(p: (E, e_0) \to (B, b_0)\) be a covering map. The induced homomorphism \(p_*: \pi_1(E, e_0) \to \pi_1(B, b_0)\) is injective. This means the fundamental group of the covering space is isomorphic to a subgroup of the fundamental group of the base space.
    \item The set of lifts of a loop in \(B\) based at \(b_0\) corresponds to the elements of the subgroup \(p_*(\pi_1(E, e_0))\). Specifically, a loop \(\gamma\) in \(B\) lifts to a loop in \(E\) if and only if its homotopy class \([\gamma]\) is in the subgroup \(p_*(\pi_1(E, e_0))\).
\end{itemize}

\item \textbf{\textbf{Classification of Covering Spaces:}} For a 'nice' space \(B\) (path-connected, locally path-connected, and semilocally simply-connected), there is a one-to-one correspondence between the subgroups of its fundamental group \(\pi_1(B, b_0)\) and the isomorphism classes of based covering spaces of \(B\). The trivial subgroup \(\{e\}\) corresponds to a special covering space called the \textbf{universal covering space}, which is simply connected (i.e., its fundamental group is trivial).

\item \textbf{From another query: How is a Covering Space Related to a subgroup of a Fundamental Group?}.  The terminology is a misnomer.  The word 'covering' implies 'larger than' as in the definition of 'compact' has 'finite subcover' and whose union is larger.  A covering space is much more related to a subgroup which IS NOT be as large as the original group.
\end{enumerate}

\item \textbf{Describe the homology of a torus.}

To describe the homology of a torus, we consider the standard 2-torus, denoted as $T^2$ or $S^1 \times S^1$. The homology groups $H_n(T^2; \mathbb{Z})$ provide information about the ``holes" of various dimensions within the shape.

\[\begin{tikzcd}[row sep=huge]
		v \arrow[r, "b"] & v \\
		v \arrow[u, "a" ] \arrow[r, swap, "b"] & 
		v \arrow[u, swap, "a"]
	\end{tikzcd}\]

\begin{description}

\item \textbf{Step-by-Step Explanation}

\begin{description}

\item 1. \textbf{Definition and CW Structure}

A torus can be constructed by taking a square and identifying opposite edges. This gives it a CW complex structure consisting of:
\begin{itemize}

\item  \textbf{One 0-cell (vertex):} Let's call it $v$.
\item  \textbf{Two 1-cells (edges):} Let's call them $a$ and $b$. Both start and end at $v$.
\item  \textbf{One 2-cell (face):} Let's call it $f$. The boundary of this face is attached along the path $aba^{-1}b^{-1}$ (the commutator of the two edges).

\end{itemize}
\item 2. \textbf{Calculating $H_0(T^2)$ (Connectedness)}

The 0\ITH homology group represents the number of path-connected components. Since the torus is a single connected piece, the 0\ITH  homology group is the free abelian group of rank 1.

\textbf{Result:} $H_0(T^2) \cong \mathbb{Z}$.

\item 3. \textbf{Calculating $H_1(T^2)$ (1-Dimensional Holes)}

The 1st homology group represents the ``loops" that cannot be shrunk to a point or expressed as boundaries of a surface. 

\begin{itemize}
\item  From the CW structure, the boundary of the 2-cell $f$ is $\partial f = a + b - a - b = 0$ in the abelian chain group.

\item  Since the boundary is zero, the two 1-cells $a$ and $b$ are cycles and are not boundaries of any 2-cell.

\item  These two cycles correspond to the ``meridional" and ``longitudinal" circles of the torus.

\item  \textbf{Result:} $H_1(T^2) \cong \mathbb{Z} \oplus \mathbb{Z}$ (or $\mathbb{Z}^2$).
\end{itemize}

\item 4. \textbf{Calculating $H_2(T^2)$ (2-Dimensional Voids)}

\item \textbf{Describe relative homology.}

\DEFINE{Relative homology} measures the homology of a space \DEFINE{relative to a subspace}—intuitively, it captures the holes of $X$ that are ``not already present in $A$".

Let’s go step by step.

---

\begin{description}

\item 1. \textbf{Setup}

Let $A \subset X$ be a subspace of a topological space $X$.

Relative homology assigns groups

$$ H_n(X,A) $$

called the \textbf{relative homology groups} of the pair $(X,A)$.

They measure how $X$ differs from $A$ topologically.

---

\item 2. \textbf{Definition using chains}

In singular homology, we start with the chain groups:

$$ C_n(X) $$

generated by singular (n)-simplices in (X).

Similarly we have

$$ C_n(A) $$

generated by simplices lying in $A$.

Since $A \subset X$, we have

$$ C_n(A) \subset C_n(X).  $$

The \textbf{relative chain group} is the quotient

$$ C_n(X,A) = C_n(X) / C_n(A).  $$

So chains in $A$ are treated as \textbf{zero}.

The boundary operator descends to this quotient, giving a chain complex

$$ \dots \to C_{n+1}(X,A) \to C_n(X,A) \to C_{n-1}(X,A) \to \dots $$

Then

$$ H_n(X,A) = \frac{\ker(\partial_n : C_n(X,A)\to C_{n-1}(X,A))} {\operatorname{im}(\partial_{n+1})}.  $$

---

\item 3. \textbf{Geometric interpretation}

Relative cycles are chains in $X$ whose boundary lies in $A$.

So:

* cycles may \textbf{end on $A$}
* boundaries are those filled inside $A$

Thus relative homology detects features of $A$ that \textbf{become trivial once $A$ is allowed as a boundary}.

---

\item 4. \textbf{Example}

Consider the disk and its boundary:

$$ X = D^2, \qquad A = S^1.  $$

Here $S^1$ is the boundary of the disk.

Key result:

$$ H_2(D^2,S^1) \cong \mathbb{Z}.  $$

Interpretation:

* The disk itself represents a \textbf{relative 2-cycle}
* Its boundary lies in $S^1$

So it is allowed in relative homology.

Note that:

$$ H_2(D^2)=0 $$

but relative homology detects the 2-dimensional region.

The boundary circle is the familiar space Circle.

---

\item 5. \textbf{Long exact sequence of a pair}

Relative homology fits into a fundamental structure:

*** Long exact sequence

$$ \cdots \to H_n(A) \to H_n(X) \to H_n(X,A) \to H_{n-1}(A) \to H_{n-1}(X) \to \cdots $$

This sequence shows how the topology of $A$, $X$, and the pair $(X,A)$ interact.

It is one of the most important tools in algebraic topology.

---

\item 6. \textbf{Intuition}

Relative homology answers the question:

 \textbf{What new holes appear when we enlarge $A$ to $X$?}

or equivalently

 \textbf{Which cycles in $X$ become trivial once $A$ is allowed as a boundary?}

---

\item 7. \textbf{Special cases}

\begin{description}

\item If $A = \emptyset$

$$ H_n(X,\emptyset) = H_n(X).  $$

So ordinary homology is a special case.

---

\item If $A = X$

$$ H_n(X,X) = 0.  $$

Because everything becomes trivial.

---

\end{description}

\item 8. \textbf{Important applications}

Relative homology is used in:

\begin{description}

\item \textbf{excision theorem}
\item \textbf{Mayer–Vietoris sequence}
\item \textbf{Poincaré–Lefschetz duality}
\item manifolds with boundary
\item CW complex computations

\end{description}

For example, manifolds with boundary use

$$ H_n(M,\partial M).  $$

\end{description}
---

\item Key intuition

Relative homology studies \textbf{chains in $X$ whose boundaries lie in $A$}.

So it measures the topology of $X$ \textbf{relative to the part $A$ that we treat as negligible}.

\item \textbf{Describe the Excision Theorem}

The \textbf{excision theorem} is one of the central results in relative homology. It tells us that \DEFINE{removing a suitable subset from both $X$ and $A$ does not change the relative homology}. This makes relative homology extremely powerful because it lets us simplify spaces without affecting the answer.

---

\begin{description}
\item 1. \textbf{Statement of the excision theorem}

Let $A \subset X$, and suppose $U \subset A$ is a subset whose closure lies in the interior of $A$:

$$ \overline{U} \subset \operatorname{int}(A).  $$ Then the inclusion

$$ (X-U, A-U) \hookrightarrow (X,A) $$

induces isomorphisms

$$ H_n(X-U, A-U) \cong H_n(X,A) $$

for all $n$.

---

\item 2. \textbf{Meaning of the theorem}

We can \textbf{cut out} a region $U$ that lies completely inside $A$ without affecting the relative homology.

So relative homology depends only on the portion of $X$ \textbf{near the boundary between $X$ and $A$}.

Everything deep inside $A$ can be removed.

---

\item 3. \textbf{Intuition}

Relative homology measures chains in $X$ whose boundaries lie in $A$.

If a region $U$ sits entirely inside $A$:

\begin{itemize}

\item  any chains inside $U$ are already considered trivial
\item  removing them does not affect the homology

Thus the topology inside $A$ is irrelevant.

Only the \textbf{interaction between $X$ and $A$} matters.

\end{itemize}
---

\item 4. \textbf{Why this theorem is useful}

Excision allows us to \textbf{shrink complicated regions} to simpler ones.

Many important computations in algebraic topology depend on it, including:

\begin{itemize}

\item homology of spheres
\item CW complex homology
\item Mayer–Vietoris sequence
\item manifolds with boundary
\end{itemize}

---

\item 5. \textbf{Classic example: computing sphere homology}

Consider the pair $(D^n, S^{n-1}) $ where $D^n$ is a disk and $S^{n-1}$ its boundary.

Using excision, one shows

$$ H_k(D^n,S^{n-1}) = \BINDEF{\mathbb{Z} & k=n }{ 0 & k\neq n } $$

This calculation helps prove the homology of spheres like the familiar boundary of a disk, the Circle.

---

\item 6. \textbf{Geometric picture}

Imagine:

\begin{itemize}

\item $A$ is a thick region inside $X$
\item $U$ is a smaller piece deep inside $A$
\end{itemize}

Excision says:

\> we can carve out $U$ completely, and the relative homology stays the same.

So the interior of $A$ becomes irrelevant.

Only the \textbf{interface between $X$ and $A$} carries homological information.

---

\item 7. \textbf{Why excision is surprising}

Ordinary homology \textbf{does not allow this}.

If you remove a subset from a space, its homology usually changes.

But relative homology is built so that:

\begin{enumerate}

\item chains in $A$ are ignored
\item therefore parts deep inside $A$ do not matter
\end{enumerate}

This special feature is exactly what excision formalizes.

---

\item 8. \textbf{Key idea behind the proof $intuition$}

The proof relies on \textbf{subdividing singular simplices}.

After enough subdivision:

* every simplex lies either inside $X-U$ or inside $A$

So chains can be replaced by equivalent chains that avoid $U$.

Thus removing $U$ does not change the homology.

---

\item 9. \textbf{Key takeaway}

The excision theorem says:

\> If a subset lies entirely inside $A$, we can remove it from both $X$ and $A$ without changing the relative homology.

This allows complicated spaces to be \textbf{simplified locally}, which is why excision is one of the most powerful tools in algebraic topology.

\end{description}
