		\item In one space dimension, Laplace's equation $\Delta u=0$ becomes an ODE
		\begin{align*}
			u''(x)=0
		\end{align*}Describe all solutions to this ODE posed on the real line $(-\infty, \infty)$.  Next, for given constants $c_1,c_2$, find the uniqe solution to $u''(x)=0$ on the interval $[0,1]$ with $u(0)=c_1$ and $u(1)=c_2$.\\
		Do these one-dimensional harmonic functions satisfy the mean value property $u(x)=\fint_{B(x,y)}u(y)dy$?  Why or why not?

		\BLUE{\begin{align*}
			\int u''(x)dx &= u'(x) + c \\
			\int (u'(x) + c)dx &= u(x)+cx+d \\
			\therefore u(x)+cx+d &= 0
		\end{align*}When $u(0)=c_1$ and $u(1) = c_2$ we have
		\begin{align*}
			u(0)+c(0)+d = 0 &\implies c_1+d=0, d=-c_1 \\
			u(1)+c(1)+d = 0 &\implies c_2+c+d=0 \\
			c_1 = c_2+c &\implies c=c_1-c_2
		\end{align*}That is
		\begin{align*}
			u(x) +(c_1-c_2)x-c_1 &=0 \\
			u(x) &= c_1-(c_1-c_2)x
		\end{align*}is our solution.  Do these one-dimensional harmonic functions satisfy the mean value property $u(x)=\fint_{B(x,y)}u(y)dy$?
		\begin{align*}
			B(x,y) &= [a,b]\\
			\fint_{B(x,y)}u(y)dy &= \frac{1}{b-a}\int_a^b u(y)dy \\
			&= \frac{1}{b-a}\int_a^b(c_1-(c_1-c_2)y) dy	\\
			&= \frac{1}{b-a}\SQBRACKET{c_1y - \frac{c_1-c_2}{2}y^2}_a^b \\
			&=\frac{1}{b-a}\SQBRACKET{c_1(b-a) - \frac{c_1-c_2}{2}(b-a)^2} \\
			&= c_1 - \frac{c_1-c_2}{2}(b-a) \\
			&= \frac{2c_1-c_1+c_2}{2} & \text{when }[a,b] = [0,1] \\
			&= \frac{c_1+c_2}{2}
		\end{align*}which is the average.		}
				
		\item Let $z=x+iy$ be a complex variable ($x$ and $y$ are real numbers).  Recall that a complex function $f(z)=u(x,y)+iv(x,y)$ is \textit{complex-differentiable} if $u$ and $v$ are continuously differentibable (as functiosn of $x$ and $y$) and Cauchy-Reimann eqations are satisfied:
		\begin{align*}
			\PART{u}{x}&=\PART{v}{y}\\
			\PART{u}{y}&=-\PART{v}{x}
		\end{align*}If $f$ is complex-differentiable, and in addition the real and imaginary parts $u(x,y)$ and $v(x,y)$ are $C^2$ (twice continuously differentiable) functions, then show that $u$ and $v$ are harmonic.

			\Delta u &= \frac{\partial^2 u}{\partial x^2}+ \frac{\partial^2 u}{\partial y^2} \\

			&= \PART{}{x}\PAREN{\PART{v}{y}}-\PART{}{y}\PAREN{\PART{v}{x}}\\

		\begin{align*}
			v(x) := u(Ox) \, (x \in \R^n)
		\end{align*}then $\Delta v=0$.
		
		\BLUE{\begin{align*}
			\Delta v &= \Delta u(Ox) \\
			Ox_j &= \sum_{k=1}^n a_{kj}x_j\\
			\PART{u(Ox_j)}{x_j} &= \sum_{k=1}^n \PART{u}{x_k}\PART{a_{kj}x_j}{x_j}\\
			&= \sum_{k=1}^n \PART{u}{x_k}a_{kj}\\
			\frac{\partial^2 u(Ox_j)}{\partial^2x_j}  &= \PART{}{x_j}\PAREN{\sum_{k=1}^n \PART{u}{x_k}a_{kj}} \\
			&= \sum_{k=1}^n \PART{}{x_j}\PAREN{\PART{u}{x_k}a_{kj}} \\
			&= \sum_{k=1}^n a_{kj}\PART{}{x_j}\PART{u}{x_k} \\
			&= \sum_{k=1}^n a_{kj} 
		\end{align*}

		\item We say $v \in C^2(\bar{U})$ is \textit{subharmonic} if
		\begin{align*}
			-\Delta v \le 0\, \text{in } U.
		\end{align*}
		\begin{enumerate}[label=(\alph*)]
			\item Prove for subharmonic $v$ that
			\begin{align*}
				v(x) \le \fint_{B(x,r)} v dy \, \text{ for all } B(x,y) \subset U.
			\end{align*}
			
			\BLUE{Remember $\Delta v = \nabla \cdot \nabla v$ and from the Divergence Theorem we have
			\begin{align*}
				\int_{B(x,r)} \nabla \cdot \nabla v\, dV &= \int_{\partial B(x,r)} \nabla v\cdot dS 
			\end{align*}by the Strong maximum principle let $b = \underset{\partial B(x,r)}\max  v$.  Then
			\begin{align*}
				\int_{\partial B(x,r)} \nabla v\cdot dS  &= \int_{\partial B(x,r)} v \,dr\\
				&\ge b|{\partial B(x,r)}| \\
				\max_{\partial B(x,r)} v &\le \frac{1}{|{\partial B(x,r)}|}\int_{\partial B(x,r)} v \,dr \\
				v(x) &\le \fint_{B(x,r)} v\, dy
			\end{align*}
			}
			\item Prove that therefore $\underset{\bar{U}}\max \,v = \underset{\partial U}\max\, v$.
			
			\BLUE{Because of the Strong Maximum Principle, the maximum cannot exist on the interior of $U$ unless $v$ is a constant.  Therefore the $\underset{\bar{U}}\max \,v = \underset{\partial U}\max\, v$
			}
			
			\item[(d)] Prove $v := \BARS{Du}^2$ is subharmonic, wherever $u$ is harmonic.
			
			\BLUE{\begin{align*}
				|Du| &= \sum_{1=1}^n (u_{x_i})^2 \\
				|Du|^2 &= \PAREN{\sum_{1=1}^n (u_{x_i})^2}^2 \\
				\PART{v}{u_{x_j}} &= \sum_{i=1}^n 2u_{x_i}u_{x_ix_j} \\
				\frac{\partial^2 }{\partial u_{x_j}^2} &= \sum_{i=1}^n 2(u_{x_ix_j}u_{x_ix_j}+u_{x_i}u_{x_ix_jx_j}) \\
				\Delta v &= \sum_{j=1}^n \sum_{i=1}^n 2(u_{x_ix_j}u_{x_ix_j}+u_{x_i}u_{x_ix_jx_j}) \\
				&= \sum_{j=1}^n \sum_{i=1}^n 2(u_{x_ix_j}^2+u_{x_i}u_{x_ix_jx_j})
			\end{align*}Since $\Delta u=0$ then all derivatives of $\Delta u$ are zero. Thus, we have
			\begin{align*}
				\Delta v &= \sum_{j=1}^n \sum_{i=1}^n 2(u_{x_ix_j})^2
			\end{align*}which is greater than zero and hence subharmonic.
			}
		
		\end{enumerate}
