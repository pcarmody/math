\item Assume $u$ is a solution for the heat equation on a bounded domain:
\begin{align*}
	u_t-\Delta u=0,\, t>0, x\in U.
\end{align*}Show that if $u$ satisfies the Dirichlet boundary condition $u=0$ on $\partial U$ OR the Neumann boundary condition $\PART{u}{\nu}=0$ on $\partial U$, then the quantity $\int_U u^2(x,t)\,dx$ is a decreasing function of time.

\BLUE{Let $v=\int_U u^2(x,t)\,dx$.  Then $S$ is decreasing if $S_t<0$ for all $t>0$.
	\begin{align*}
		v_t &= \int_U 2uu_t \,dx \\
		&= \int_U 2u\Delta u\, dx
	\end{align*}Using Integration by Parts
	\begin{align*}
		\int_U 2u\Delta u\, dx &= -2\int_U |Du|^2\,dx +2 \int_{\partial U} u\frac{du}{d\nu}\, dS
	\end{align*}Thus, the second term is zero when $u=0$ on $\partial U$ or when $\PART{u}{\nu}=0$ on $\partial U$.  The integral of the first time is positive as $|Du|^2$ is positive and the minus sign makes the expression negative.  Hence, $v_t< 0$ for all $t>0$.
}

\item Consider the heat equation on a bounded domain with Newmann boundary conditions on $\partial U$:
\begin{align*}
	\BINDEF{u_t-\Delta u=0, & x \in U, t>0}{\PART{u}{\nu}=0,& x \in \partial U}
\end{align*}Show that the quantity $\int_U\BARS{u(x,t)}^2\,dx$ is decreasing as a function of time.

\BLUE{Keeping in mind that the value of $u(x,t)$ could be complex, then $|u(x,t)|^2 = u\CONJ{u}$.  Then,
\begin{align*}
	\LET v = \int_U|u(x,t)|^2\,dx &= \int_U u(x,t)\CONJ{u(x,t)}\, dx \\
	\frac{dv}{dt} &= \int_U \PAREN{u_t\CONJ{u} + u\CONJ{u}_t}\, dx \\
	\frac{dv}{dt} &= \int_U \PAREN{\Delta u\CONJ{u} + u\Delta \CONJ{u}}\, dx 
\end{align*}Applying Integration by Parts as in 2.  We have
\begin{align*}
	v_t &= \int_U \Delta u\CONJ{u}\,dx + \int_U u\Delta \CONJ{u}\, dx \\
	&= -\int_U Du\cdot D\CONJ{u} \,dx + \int_{\partial U} u\CONJ{u}_t\, dx - \int_U D\CONJ{u}\cdot Du \,dx + \int_{\partial U} \CONJ{u}u_t\, dx
\end{align*}The two boundary integrals are zero because of the Newman Boundary condition.  Since the dot product is commutative, we have
\begin{align*}
	v_t &= -2\int_U Du\cdot D\CONJ{u} \,dx = -2\int_U |Du|^2 \, dx 
\end{align*}Thus, $v_t < 0$ for all $t>0$ therefore decreasing.
}

Recall that solutions to Neumann problem for Laplace's equations were not unique.  Why doesn't the same problem occur here?

\BLUE{Let $u,v$ be solutions to the Neumann problem for the inhomogeneous heat equation.  Let $w = u-v$. Then,
\begin{align*}
	w &= u - v \AND \PART{w}{\nu} = \PART{u}{\nu}-\PART{v}{\nu} = h - h = 0\\
	\int_U w^2 dx &= \int_U w\Delta w\, dx \\
	&= -\int_U |Dw|^2\, du + \int_{\partial U} w \PART{w}{\nu}, Ds\\
	&= -\int_U |Du-Dv|^2\, dx \\
	&= -\int_U (|Du|^2 + |D^v|^2-2Du\cdot Dv)\, dx & (*)\\
\end{align*}Notice that at $t=0$ we have
\begin{align*}
	g'(x,0)-\Delta u &= g'(x,0)-\Delta v) \\
	\Delta u &= \Delta v 
\end{align*}Integrate both sides and we have $Du = Dv$ then from (*)
\begin{align*}
	-\int_U (|Du|^2 + |D^v|^2-2Du\cdot Dv)\, dx &= -\int_U (|Du|^2 + |D^v|^2-2|Du|^2)\, dx \\
	&= 0
