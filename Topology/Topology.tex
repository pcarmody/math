\begin{enumerate}

\item Prove that if $a,b \in \R$ with $a < b$ then neither $[a,b)$ nor $(a,b]$ is an open subset of $\R$.  Also show that neither is a closed subset of $\R$.

In the case of $[a,b)$ there is no set $a \in (x,y)$ because $x < a$ implies that $x+\frac{|x-a|}{2}$ would have to be a member of $[a,b)$ which it cannot.  Similarly for $(a,b]$.

\item Prove that the sets $[a,\infty)$ and $(-\infty, a]$ are closed subsets of $\R$.

The composite of $[a,\infty)$ is $(-\infty, a)$ which is open and similarly for $(-\infty, a]$.

\item Show, by example, that the union of an infinite number of closed subsets of $\R$ is not necessarily a closed subset of $\R$.

Define $S_i = [1/i, 1]$ then $\mathcal{S} = \cup_{i=1}^\infty S_i$.  Obviously, given any $n \in \N$ there is a closed set $S_n=[1/n, 1]$ and there exists $(1/(n+1),1) \subseteq \mathcal{S}$ such that $1/n \in  (1/(n+1),1)$ hence $\mathcal{S}$ must be open.

\section{Basic and T-spaces}

\begin{definition}[Discrete/Indiscrete Topologies]Let $X$ be any non-empty set and let $\TT$ be the collection of all subsets of $X$ . Then topological space ($X$, $\TT$) is called \DEFINE{the discrete topology} on the set $X$.\\
\\
Let X be any non-empty set and is called \DEFINE{the indiscrete topology} and $(X, \TT) \to \TT = \{X, \emptyset\}$.
\end{definition}

\begin{definition}[Finite-Closed Topologies]Let $X$ be any non-empty set. A topology $\TT$ on $X$ is called the \DEFINE{finite-closed topology or the cofinite topology} if the closed subsets
of $X$ are $X$ and all finite subsets of $X$ ; that is, the open sets are $\emptyset$ and all subsets of $X$ which have finite complements.
\end{definition}

\begin{definition}[Countable-Closed Topologies]
\end{definition}

\begin{definition}[$T_0$ Spaces]A topological space $(X,\TT)$ is said to be a \textbf{$T_0$-space} if for each pair of distinct points $a,b$ in $X$, either there exist an open set containing $a$ and not $b$, or there exists an open set containing $b$ and not $a$.
\end{definition}

\begin{definition}[$T_1$ Spaces]A topological space $(X, \TT)$ is said to be a \textbf{$T_1$-space} if every singleton set $\{x\}$ is closed in $(X,\TT)$.
\end{definition}

\begin{definition}[$T_2$ Space or Hausdorf]A topological space $(X,
\TT )$ is said to be Hausdorff (or a $T_2$-space) if given any
pair of distinct points $a, b$ in $X$ there exist open sets $U$ and $V$ such that $a \in U, b \in V,$ and $U \cap V = \emptyset$ .
\end{definition}

\begin{definition}[$T_3$ Spaces, Regular Space]A topological space $(X,
\TT )$ is said to be a regular space if for any closed subset
$A$ of $X$ and any point $x \in X \backslash A$, there exist open sets $U$ and $V$ such that $x \in U, A \subseteq V$, and $U \cap V = \emptyset$ . If $(X, \TT )$ is regular and a $T_1$-space, then it is
said to be a $T_3$-space
\end{definition}

\section{Countability and Topologies}

\begin{definition}[Countable-closed topology] Let $X$ be any infinite set.  The \textbf{\textit{countable-closed topology}} is defined to be the topology having as its closed sets $X$ and all countable subsets of $X$.
\end{definition}

\begin{definition}[Second Axiom of Countability]A topological space $(X, \TT)$ is said to satisfy the \DEFINE{second axiom of countability} or to be \DEFINE{second countable} if there exists a basis $\BB$ for $\TT$, where $\BB$ consists of only a countable number of sets.
\end{definition}

\begin{definition}[$F_\sigma$ set]Let $(X, \TT)$ be a topological space.  \begin{center} A subset $S$ of $X$ is said to be an \DEFINE{$F_\sigma$ set} if it is the \textbf{union of a countable number of closed sets}.\end{center}
\end{definition}

\begin{definition}[$G_\delta$-set]Let $(X, \TT)$ be topological space.  \begin{center}A subset $T$ of $X$ is said to be a \DEFINE{$G_\delta$-set} if it is the \textbf{intersection of a countable number of open sets}.\end{center}
\end{definition}

\begin{definition}[Product Topology]Let $\BB_1$ be a basis for a topology $\TT_1$ on a set $X$ and $\BB_2$ a basis for a topology $T_2$ on a set $Y$.  the set $X \times Y$ consists of all ordered pairs $\ABRACKET{x,y}, x \in X$ and $y \in Y$.  Let $\BB$ be the collection of subsets of $X \times Y$ consisting of all the sets $B_1 \times B_2$ where $B_1 \in \BB_1$ and $B_2 \in \BB_2$.  Prove that $\BB$ is a basis for a topology on $X \times Y$.  the topology so defined is called the \DEFINE{product topology} on $X \times Y$.
\end{definition}

\begin{definition}[Subbasis] Let $(X, \TT)$ be a topological space.  A non-empty collection $\SB$ of open subsets of $X$ is said to be \DEFINE{subbasis} of $\TT$ if the collection of all finite intersections fo members of $\SB$ forms a basis for $\TT$.
\end{definition}

\begin{definition}[Door Space] A topological space $(X,T)$ is said to be a \textbf{\textit{door space}} if every subset of $X$ is either an open set or a closed set (or both)
\end{definition}

\begin{definition}[Saturated Topological Space]A subset $S$ of a topological space $(X,\TT)$ is said to be \textbf{saturated} if it is an intersection of open sets in $(X, \TT)$.
