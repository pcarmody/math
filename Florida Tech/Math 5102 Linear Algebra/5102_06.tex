	\item Let $V$ and $W$ be $n$-dimensional vector spaces, and let $T: V \to W$ be a linear transformation.  Suppose that $\beta$ is a basis for $V$.  Prove that $T$ is an isomorphism if and only if $T(\beta)$ is a basis for $W$.
	
	\BLUE{\begin{description}
		\item $(\Longrightarrow)$ $T$ is an isomorphism, thus $T$ is one-to-one and onto.  Thus, 
		\begin{align*}
			\forall w \in W, \,&!\exists v \in V \to v = \sum_{i=1}^n \alpha_i\beta_i \AND T(v) = w\\
			w &= T(v) = T\PAREN{\sum_{i=1}^n \alpha_i\beta_i} = \sum_{i=1}^n \alpha_iT(\beta_i) \\
			\text{also } \forall i,j, i\ne j, 0&= aT(\beta_i) - bT(\beta_j) = T(a\beta_i - b\beta_j) \to a=b=0 \\
			\therefore \SPAN\BRACKET{T(\beta_i)} &= W
		\end{align*}
		\item $(\Longleftarrow)$ $\BRACKET{T(\beta_i)}$ is a basis for $W$.
		\begin{align*}
			\dim \SPAN\BRACKET{T(\beta_i)} &= n \to \text{onto}
		\end{align*}
	\end{description}
	}

\end{enumerate}
